課題8　解答例


dict = {'item1':45.50,'item2':3.5,'item3':41.30,'item4':55,'item5':24}
array = []

for k, v in sorted(dict.items(), key=lambda x: -x[1])[0:3]:
    array.append(str(k))
    if len(array) > 4:
        break
array = sorted(array)
print(array)

課題8　解説


#辞書型
#リストは要素の格納順で番号が振られた。(例えばlist[2]ならば2番目の要素を対象etc..)
#辞書型はキーとバリューの組み合わせで格納していく。
辞書名 = { キー値1 : バリュー値1 , キー値2 : バリュー値2 ...}
#みたいな感じで使う

#例えば...
dict = {"赤いもの":"りんご","黄色いもの":"バナナ","かわいい人":"吉岡里帆"}

print(dict["赤いもの"])
print(dict["黄色いもの"])
print(dict["かわいい人"])


コンソール結果：
りんご
バナナ
吉岡里帆

#key,valueはint型でもstr型でも平気!(上記の例はstr:str,課題はstr:int)

#今回の問題も前回と同様新しい配列を作り、その配列に大きい順で詰めていくようなイメージ
#dictをvalue値で降順で並び替え、上から上位３つをarrayに詰めていく方法をとった
#key=lambda x: -x[1] の箇所がソート、
#[0:3]の箇所がステップ

dict = {'item1':45.50,'item2':3.5,'item3':41.30,'item4':55,'item5':24}
array = [] #この配列に値を詰めていく

for k, v in sorted(dict.items(), key=lambda x: -x[1])[0:3]:     #ソートの順番をいじりたいので
    array.append(str(k))
    if len(array) > 4: 
        break
array = sorted(array)   #これをしないと解答例のような昇順にはならない
print(array)
