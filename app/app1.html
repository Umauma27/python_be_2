1.pandas

データ分析を行う際の様々な便利な機能を提供されているライブラリ。データ分析を行う上で用いる!
PandasにはDataFrameとSeriesの2種類のデータ形式がある

1.1 series


# numpyのインポート
import numpy as np
#pandas, series,DataFrameのインポート
from pandas import Series, DataFrame
import pandas as pd

even = Series([2,4,6,8])
print(even)
コンソール結果：
0    2
1    4
2    6
3    8
dtype: int64     #どんな型かも出力される




print(even.values)
コンソール結果：
[2 4 6 8]



#index付きのデータを作る
#果物の価格
fruits = Series([250,200,150],index=['grape','orange','apple'])
print(fruits)

コンソール結果：
grape     250
orange    200
apple     150
dtype: int64


# 文字列のindexでアクセスできる
print(fruits['apple'])

コンソール結果：
150


#辞書型似てる....?
#そう！辞書型に変換できる！
fruits_dict = fruits.to_dict()
print(fruits_dict)

コンソール結果：
{'grape': 250, 'orange': 200, 'apple': 150}


#逆も可能!
pico_dict = {'apple': 350, 'pineapple': 500, 'pen': 250}
pico_series = Series(pico_dict)
print(pico_series)

コンソール結果：
apple        350
pineapple    500
pen          250
dtype: int64

#Dataflameはデータベースのように「excelの表」みたいな感じで扱える！

# 各々モジュールのインポート忘れずに!!
import numpy as np
from pandas import Series, DataFrame
import pandas as pd


df1 = pd.DataFrame(
{'age': [29, 24, 29, 10, 5, 0],
 'sex': ['m', 'f', 'm', 'm', 'f', 'f' ],'tall':[172,160,180,175,153,158]},
 columns=['age', 'sex','tall'],
 index=['Mike', 'Jane', 'Joe', 'Dan', 'Steph', 'Michelle'])

 print(df1)
コンソール結果：
          age sex  tall
Mike       29   m   172
Jane       24   f   160
Joe        29   m   180
Dan        10   m   175
Steph       5   f   153
Michelle    0   f   158

#カラムの検索「.columns」
df1.columns
コンソール結果：
Index(['age', 'sex', 'tall'], dtype='object')

#jupyterの場合、print()を使わないで出力してみよう!
DataFrame(df1,columns=['age','tall'])


#列の追加「データフレーム名['インデックス名']=[データ1,データ2,...]」
df1['weight']=[60,45,70,75,50,30] 
print(df1)

コンソール結果：

          age sex  tall  weight
Mike       29   m   172      60
Jane       24   f   160      45
Joe        29   m   180      70
Dan        10   m   175      75
Steph       5   f   153      50
Michelle    0   f   158      30

#同様に行の追加も(こちらは1行のデータフレームを作成してから.appendで挿入するイメージ)
